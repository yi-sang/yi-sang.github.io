import sys

def BinarySearch(arr,target):
    L, R = 0, len(arr)-1
    while(L <= R):
        M = (L + R) // 2
        if(arr[M] == target):
            return 1
        elif(arr[M] > target):
            R = M - 1
        else:
            L = M + 1
    return 0
